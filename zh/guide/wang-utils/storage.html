# 设置storage（setStore）

<template>
  <div>
    <el-input v-model="value"></el-input>
    <el-button @click="setStore">设置storage</el-button>
  </div>
</template>
<script>
export default {
  data () {
    return {
      value: '123'
    }
  },
  methods: {
    setStore () {
      const content = this.value
      WangUtils.setStore({ name: 'name1', content })
    }
  }
}
</script>

# 读取单个storage（getStore）

  </div>
</template>
<script>

  methods: {
    getStore () {

    }
  }
}
